//criar personagem
function criarPersonagem(player) = player(classe, lvl) {
  return {
    nome: classe.nome,
    lvl,
    xp: 0,
    xpToUp: 100,
    hpMax: classe.hp(lvl),
    hp: classe.hp(lvl),
    def: classe.def(lvl),
    atk: classe.atk 
    }
  };

//ganhoXP
function XpUP(personagem, quantidade) {
  personagem.xp += quantidade;
  console.log(`${personagem.nome} ganhou ${quantidade} XP`);

  if (personagem.xp >= personagem.xpToUp) {
    lvlUP(personagem);
  } 
};

//lvlUP
function lvlUP(personagem) {
  personagem.xp -= personagem.xpToUp;
  personagem.lvl++;

  personagem.xpToUp = Math.floor(personagem.xpToUp * 1.5);

  personagem.hpMax += d10();
  personagem.hp = personagem.hpMax;
  
  console.log(`${personagem.nome} subiu para o lvl ${personagem.lvl}!`);
};
